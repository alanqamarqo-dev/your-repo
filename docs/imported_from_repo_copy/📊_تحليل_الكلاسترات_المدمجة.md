```python
from Integration_Layer.meta_orchestrator import MetaOrchestrator
from Integration_Layer.inproc_bus import PriorityBus
from Integration_Layer.knowledge_graph import KnowledgeGraph as DKNGraph
from Integration_Layer.engine_adapter import EngineAdapter

def __init__(self, engine_registry):
    # ... الكود الحالي ...
    
    # إضافة DKN
    self.dkn_graph = DKNGraph()
    self.priority_bus = PriorityBus()
    
    # تحويل المحركات لـ adapters
    self.engine_adapters = []
    for name in ['Creative_Innovation', 'Mathematical_Brain', 
                 'Self_Reflective', 'Causal_Graph', 'Reasoning_Layer']:
        engine = engine_registry.get(name)
        if engine:
            adapter = EngineAdapter(
                name=name,
                engine=engine,
                subscriptions=['task:new'],
                capabilities=['process']
            )
            self.engine_adapters.append(adapter)
    
    # المنسق الذكي
    self.meta_orchestrator = MetaOrchestrator(
        self.dkn_graph,
        self.priority_bus,
        self.engine_adapters
    )
```

```python
    VisionAdapter,

def __init__(self, engine_registry):

    self.perception_bus = PerceptionBus()
    
    # المحولات الحسية
    self.vision_adapter = VisionAdapter()
    self.audio_adapter = AudioAdapter()
    self.sensor_adapter = SensorAdapter()
    self.perceptual_integration = PerceptualIntegrationAdapter()

# إضافة دالة جديدة
async def process_multimodal(self, input_data: Dict) -> Dict:
    """معالجة وسائط متعددة (نص + صورة + صوت)"""
    
    results = {}
    
    # معالجة الصورة إن وجدت
    if 'image' in input_data:
        vision_result = self.vision_adapter.process(input_data['image'])
        self.perception_bus.push('vision', vision_result, time.time())
        results['vision'] = vision_result
    
    # معالجة الصوت إن وجد
    if 'audio' in input_data:
        audio_result = self.audio_adapter.process(input_data['audio'])
        self.perception_bus.push('audio', audio_result, time.time())
        results['audio'] = audio_result
    
    # معالجة الحساسات إن وجدت
    if 'sensor_data' in input_data:
        sensor_result = self.sensor_adapter.process(input_data['sensor_data'])
        self.perception_bus.push('sensor', sensor_result, time.time())
        results['sensor'] = sensor_result
    
    # دمج الإدراك
    integrated = self.perceptual_integration.integrate(results)
    
    return integrated
```

```python
async def evolve_system(self, feedback: Dict) -> Dict:
    """تطوير وتحسين النظام بناءً على الأداء"""
    
    # مراقبة الأداء
    performance = self.self_monitoring.analyze_performance(feedback)

    for improvement in improvements:
        # حفظ نقطة استعادة
        checkpoint = self.auto_rollback.create_checkpoint()
        
        try:
            # تطبيق التحسين
            result = self.safe_modification.apply(improvement)
            
            # اختبار
            if not self.self_monitoring.validate(result):
                # التراجع إذا فشل
                self.auto_rollback.restore(checkpoint)
            else:
                # حفظ في الذاكرة الاستراتيجية
                self.strategic_memory.store_successful_improvement(
                    improvement, result
                )
        except Exception as e:
            # تراجع تلقائي عند الخطأ
            self.auto_rollback.restore(checkpoint)
    
    return {"improvements_applied": len(improvements)}
```
